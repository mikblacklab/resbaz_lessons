Nesting

hourly_delay <- filter(
  summarise(
    group_by(
      filter(
        flights,
        !is.na(dep_delay)
      ),
      year, month, day
    ),
    delay = mean(dep_delay),
    n=n()
  ),
  n > 10
)

The pipe operator: %>%

  • Read the pipe as "then": x %>% f(y) -> f(x,y)
hourly_delay <- flights %>%
  filter(!is.na(dep_delay)) %>%
  group_by(year, month, day) %>%
  summarise(
    delay = mean(dep_delay),
    n = n()
  ) %>%
  filter(n > 10)
